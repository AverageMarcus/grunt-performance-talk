HTMLMin

Install using: 

npm install grunt-contrib-htmlmin --save-dev

Example Gruntfile.js: 


htmlmin: {
  dist: {
    options: {
      collapseWhitespace: true, // Remove excess spaces
      removeComments: true, // Remove all comments
      removeEmptyAttributes: true,
      removeRedundantAttributes: true, // Remove attributes with default values
      removeScriptTypeAttributes: true,
      removeStyleLinkTypeAttributes: true,
      minifyJS: true, // In script tags
      minifyCSS: true // In style tags
    },
    files: [{
      expand: true, // Enable dynamic expansion
      cwd: '_site/', // Where to search for files
      src: ['*.html','**/*.html'], // What files to minify
      dest: '_site/' // Where to save minified files (overwrite)
    }]
  }
}
